YUG_croatian_up_effect = {
	if = {
		limit = {
			has_idea = YUG_croatian_opposition_1
		}
		swap_ideas = {
			remove_idea = YUG_croatian_opposition_1
			add_idea = YUG_croatian_opposition_2
		}
	}
	else_if = {
		limit = {
			has_idea = YUG_croatian_opposition_2
		}
		swap_ideas = {
			remove_idea = YUG_croatian_opposition_2
			add_idea = YUG_croatian_opposition_3
		}
	}
	else_if = {
		limit = {
			has_idea = YUG_croatian_opposition_3
		}
		swap_ideas = {
			remove_idea = YUG_croatian_opposition_3
			add_idea = YUG_croatian_opposition_4
		}
	}
	else_if = {
		limit = {
			has_idea = YUG_croatian_opposition_4
		}
		swap_ideas = {
			remove_idea = YUG_croatian_opposition_4
			add_idea = YUG_croatian_opposition_5
		}
		if = {
			limit = { NOT = { has_global_flag = YUG_croatian_uprising_happened } }
			custom_effect_tooltip = YUG_croatian_uprising_tt
		}
	}
	else_if = {
		limit = {
			has_idea = YUG_croatian_opposition_5
		}
		add_stability = -0.1
	}
	else = {
		add_ideas = {
			YUG_croatian_opposition_1
		}
	}
}

YUG_change_name_of_state_or_vp = {
	if = { # Trieste / Trst
		limit = {
			FROM.FROM = {
				state = 736 #LITTORALE
			}
		}
		if = { #SLAVIC COUNTRY CONTROLLER
			limit = {
				ROOT = {
					OR = {
						original_tag = YUG
						original_tag = SLV
						original_tag = CRO
					}
				}
			}
			FROM.FROM = { set_state_name = STATE_736_YUG }
			set_province_name = { id = 6626 name = VICTORY_POINTS_6626_YUG }
		}
		else_if = { #OTHER CONTROLLER
			limit = {
				FROM = {
					OR = {
						original_tag = YUG
						original_tag = SLV
						original_tag = CRO
					}
				}
			}
			reset_province_name = 6626
		}
	}
	else_if = { # Fiume / Rijeka
		limit = {
			FROM.FROM = {
				state = 974 #FIUME
			}
		}
		if = { #ITALY CONTROLLER
			limit = {
				ROOT = { original_tag = ITA }
			}
			set_province_name = { id = 11564 name = VICTORY_POINTS_11564_ITA }
		}
		else_if = { #OTHER CONTROLLER
			limit = {
				FROM = { original_tag = ITA }
			}
			reset_province_name = 11564
			FROM.FROM = { set_state_name = STATE_974_YUG }
		}
	}
	else_if = { # Pola / Pula
		limit = {
			FROM.FROM = {
				state = 852 #FIUME
			}
		}
		if = { #SLAVIC COUNTRY CONTROLLER
			limit = {
				ROOT = {
					OR = {
						original_tag = YUG
						original_tag = SLV
						original_tag = CRO
					}
				}
			}
			set_province_name = { id = 11735 name = VICTORY_POINTS_11735_YUG }
		}
		else_if = { #OTHER CONTROLLER
			limit = {
				FROM = {
					OR = {
						original_tag = YUG
						original_tag = SLV
						original_tag = CRO
					}
				}
			}
			reset_province_name = 11735
		}
	}
}
