#### ECONOMIC CRISIS EFFECTS ####

HUN_check_crisis_icon = { #If certain value is reached then icon is changed, but actually the whole modifier is just switched 
	if = {	### SET IT TO SECOND AFTER LAST
		limit = {
			has_dynamic_modifier = { modifier = HUN_economic_crisis_3 }
			check_variable = {
				var = HUN_crisis_consumer_goods_factor
				value = 0.10
				compare = less_than_or_equals
			}
			check_variable = {
				var = HUN_crisis_production_speed_buildings_factor
				value = -0.15
				compare = greater_than_or_equals
			}
			check_variable = {
				var = HUN_crisis_industrial_capacity_factory
				value = -0.15
				compare = greater_than_or_equals
			}
		}
		remove_dynamic_modifier = { modifier = HUN_economic_crisis_3 }
		add_dynamic_modifier = { modifier = HUN_economic_crisis_2 }
		set_country_flag = HUN_economic_crisis_2_flag 
    }
	else_if = {	### SET IT TO NEARLY GONE
		limit = {
			has_dynamic_modifier = { modifier = HUN_economic_crisis_2 }
			
			check_variable = {	### CONSUMER GOODS
				var = HUN_crisis_consumer_goods_factor
				value = 0.05
				compare = less_than_or_equals
			}
			
			check_variable = {	### CONSTRUCTION SPEED 
				var = HUN_crisis_production_speed_buildings_factor
				value = -0.10
				compare = greater_than_or_equals
			}
			
			check_variable = {	### FACTORY OUTPUT
				var = HUN_crisis_industrial_capacity_factory
				value = -0.10
				compare = greater_than_or_equals
			} 
		}
		
		remove_dynamic_modifier = { modifier = HUN_economic_crisis_2 }
		clr_country_flag = HUN_economic_crisis_2_flag
		add_dynamic_modifier = { modifier = HUN_economic_crisis_1 }
		set_country_flag = HUN_economic_crisis_1_flag
    }
	#else_if = {	### REMOVE MODIFIER
		#limit = {
			#has_dynamic_modifier = { modifier = HUN_economic_crisis_1 }
			### CONSUMER GOODS
			#check_variable = {
			#	var = HUN_crisis_consumer_goods_factor 
			#	value = 0
			#	compare = equals
			#}
			### CONSTRUCTION SPEED
			#check_variable = {
			#	var = HUN_crisis_production_speed_buildings_factor
			#	value = 0
			#	compare = equals
			#}
			### FACTORY OUTPUT 
			#check_variable = {
			#	var = HUN_crisis_industrial_capacity_factory
			#	value = 0
			#	compare = equals
			#} 
		#}
		#country_event = LH.475  
	#}
}

### ADD CRISIS VALUES ###

HUN_add_1_consumer = {
	if = {
		limit = {
			check_variable = {
				var = HUN_crisis_consumer_goods_factor 
				value = 0
				compare = greater_than
			} 
		}
		custom_effect_tooltip = HUN_economic_crisis_1_consumer_goods
		add_to_variable = { var = HUN_crisis_consumer_goods_factor value = -0.01 } 
		clamp_variable = { var = HUN_crisis_consumer_goods_factor min = 0 }
		HUN_check_crisis_icon = yes
	}
	else_if = {
		limit = { 
			has_country_flag = HUN_economic_crisis_good_flag 
			check_variable = {
				var = HUN_good_crisis_consumer_goods_factor 
				value = -0.1
				compare = greater_than
			} 
		}
		custom_effect_tooltip = HUN_economic_crisis_1_consumer_goods
	}
	add_to_variable = { var = HUN_good_crisis_consumer_goods_factor value = -0.01 } 
	clamp_variable = { var = HUN_good_crisis_consumer_goods_factor min = -0.1 }
}
