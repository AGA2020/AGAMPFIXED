# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes

defined_text = { 
	name = Getbulwark_of_CatholicismName #HUNGARY
	text = {
		trigger = {
			original_tag = HUN
			OR = {
				has_country_leader = {
					name = "Otto von Habsburg"
					ruling_only = yes
				}
				has_country_leader = {
					name = "Albrecht Franz von Habsburg-Teschen"
					ruling_only = yes
				}
			}
			
		}
		localization_key = HUN_BULWARK_OF_CATHOLICISM_NAME_HABSBURG_desc
	}
	text = {
		trigger = {
			original_tag = HUN
			NOT = {
				has_country_leader = {
					name = "Otto von Habsburg"
					ruling_only = yes
				}
			}
		}
		localization_key = HUN_BULWARK_OF_CATHOLICISM_NAME_MONARCHY_desc
	}
}
