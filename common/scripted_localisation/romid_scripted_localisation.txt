# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes

defined_text = {
	name = GetArmyLoyaltyStatus
	text = {
		trigger = {
			check_variable = {
				var = ROMID_army_loyalty
				value = 15
 				compare = less_than_or_equals
			}
		}
		localization_key = ROMID_army_loyalty_status_amount_disloyal
	}
	text = {
		trigger = {
			check_variable = {
				var = ROMID_army_loyalty
				value = 16
 				compare = greater_than_or_equals
			}
			check_variable = {
				var = ROMID_army_loyalty
				value = 30
 				compare = less_than_or_equals
			}
		}
		localization_key = ROMID_army_loyalty_status_amount_low
	}
	text = {
		trigger = {
			check_variable = {
				var = ROMID_army_loyalty
				value = 31
 				compare = greater_than_or_equals
			}
			check_variable = {
				var = ROMID_army_loyalty
				value = 65
 				compare = less_than_or_equals
			}
		}
		localization_key = ROMID_army_loyalty_status_amount_med
	}
	text = {
		trigger = {
			check_variable = {
				var = ROMID_army_loyalty
				value = 66
 				compare = greater_than_or_equals
			}
		}
		localization_key = ROMID_army_loyalty_status_amount_high
	}
}
