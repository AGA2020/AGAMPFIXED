add_namespace = LH_FRA

#Hungary asks for help in economic crisis
country_event = {
	id = LH_FRA.1
	title = LH_FRA.1.t
	desc = LH_FRA.1.d 
	
	fire_only_once = no
	
	is_triggered_only = yes
	
	picture = GFX_report_event_generic_conference	
	
	option = {# We will help
		name = LH_FRA.1.a
		
		ai_chance = {
			factor = 30
			modifier = {
				has_opinion = { target = HUN value > 40} 
				factor = 2
			}
			modifier = {
				has_opinion = { target = HUN value > 0} 
				factor = 2
			}
			modifier = {
				is_in_faction_with = HUN
				factor = 2
			}
			modifier = {
				GER =  { check_variable = { var = ger_alignment_influence value = 15 compare = less_than_or_equals } }
				factor = 2
			}
		}
		
		HUN = {
			country_event = { id = LH.35 }
			
		}
	
		add_political_power = 100
	
	}
	
	option = {# We have no interest in such things
		name = LH_FRA.1.b
		
		ai_chance = {
			factor = 40
			modifier = {
				has_opinion = { target = HUN value < 0} 
				factor = 5
			}
			modifier = {
				has_opinion = { target = HUN value < 50} 
				factor = 2
			}
			modifier = {
				HUN = { 
					OR = {
						has_completed_focus = HUN_renounce_trianon 
						has_completed_focus = HUN_renounce_trianon_monarch
					}
				}
				factor = 5
			}
			modifier = {
				GER =  { check_variable = { var = ger_alignment_influence value = 15 compare = greater_than_or_equals } }
				factor = 2
			}
		}
		
		HUN = {
			country_event = { id = LH.36 }
		}
	}
}
